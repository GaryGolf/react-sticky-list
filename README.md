#### Usage

```jsx
import  ScrollList from 'react-scroll-list';
  return (
    <ScrollList>
      <header>Soft Dirinks</header>
      <div>Cola</div>
      <div>cold Tea</div>
    </ScrollList>
  );
```

#### Attributes

#### <code><i>boolean</i> <a name="genericscrollbox-nativescroll"></a> shadows</code>
